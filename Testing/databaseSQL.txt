CREATE TABLE derived_images(
       record_id INTEGER PRIMARY KEY AUTOINCREMENT,
       analysis VARCHAR(100) NOT NULL,
       project VARCHAR(30) NOT NULL,
       subject CHAR(4) NOT NULL,
       session CHAR(5) NOT NULL,
       location VARCHAR(500) UNIQUE NULL,
       status VARCHAR(10) DEFAULT 'unreviewed'
);

CREATE TABLE image_reviews(
       review_id INTEGER PRIMARY KEY AUTOINCREMENT,
       record_id INTEGER NOT NULL,
       caudateRight INTEGER DEFAULT NULL,
       caudateLeft INTEGER DEFAULT NULL,
       hippocampusRight INTEGER DEFAULT NULL,
       hippocampusLeft INTEGER DEFAULT NULL,
       putamenRight INTEGER DEFAULT NULL,
       putamenLeft INTEGER DEFAULT NULL,
       thalamusRight INTEGER DEFAULT NULL,
       thalamusLeft INTEGER DEFAULT NULL,
       FOREIGN KEY (record_id) REFERENCES derived_images(record_id)
);
