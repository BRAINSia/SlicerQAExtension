CREATE TABLE session
    (id INT(9) NOT NULL,
     project VARCHAR(30) NOT NULL,
     subject CHAR(4) NOT NULL,
     session CHAR(5) NOT NULL,
     experiment VARCHAR(100) NOT NULL,
     status VARCHAR(12) NOT NULL DEFAULT 'needs review',
     caudateRight INT,
     caudateLeft INT,
     hippocampusRight INT,
     hippocampusLeft INT,
     putamenRight INT,
     putamenLeft INT,
     thalamusRight INT,
     thalamusLeft INT,
     PRIMARY KEY (id)
     );
